library(animation)
library(dplyr)
library(png)
par(mar = c(1, 1, 1, 1))
ani.options(ani.dev = new.png, interval = 0.02, nmax = 200, 
    ani.width = 900, ani.height = 600)
ROCKS <- atomizeString("INGER[space]AND[space]ANDY[newline]ROCK")
for (i in 1:20) {
    dev.hold()
    plot(ROCKS, xlim = c(-7, 7), ylim = c(-1, 2))
    ani.pause()
}
jazz <- morph(ROCKS, "OTHERS[space]ARE[newline]MISTER[space]SPOCK")
returnHome(jazz, xlim = c(-7, 7), ylim = c(-1, 2), Duration = "smallVarMedium", 
    easing = "easeInQuad")
