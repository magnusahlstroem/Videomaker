##   using the package animation
library(animation)
par(mar = c(1, 1, 1, 1))
ani.options(ani.dev = png.cust, interval = 0.05, nmax = 200, 
    ani.width = 800, ani.height = 800)
HEY <- atomizeString("HEY")
for (i in 1:20) {
    dev.hold()
    plot(HEY, xlim = c(-2, 2), ylim = c(-1.5, 2.5))
    ani.pause()
}
ANDY <- morph(HEY, "ANDY")
ANDY <- returnHome(ANDY, xlim = c(-2, 2), ylim = c(-1.5, 
    2.5))
