##   using the package animation
library(animation)
par(mar = c(1, 1, 1, 1))
ani.options(interval = 0.02, nmax = 200, ani.width = 800, 
    ani.height = 800)
HEY <- atomizeString("HEY")
for (i in 1:20) {
    dev.hold()
    plot(HEY, xlim = c(-3, 3), ylim = c(-1.5, 2.5))
    ani.pause()
}
THEIS <- morph(HEY, "THEIS")
THEIS <- returnHome(THEIS, xlim = c(-3, 3), ylim = c(-1.5, 
    2.5))
for (i in 1:20) {
    dev.hold()
    plot(THEIS, xlim = c(-3, 3), ylim = c(-1.5, 2.5))
    ani.pause()
}
Smiles <- morph(THEIS, "[Smiley1][Smiley1]")
